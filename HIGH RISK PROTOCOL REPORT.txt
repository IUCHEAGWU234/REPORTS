CREATE TABLE PND_ENEH NOLOGGING NOCACHE PARALLEL(DEGREE 200) AS 
SELECT a.AC_DESC, a.CUST_AC_NO, a.CUST_NO, b.REMARKS, a.AC_STAT_NO_DR AS PND_STAT, a.LCY_CURR_BALANCE
FROM STTM_CUST_ACCOUNT a, oallen.pnd_reason b 
WHERE a.CUST_AC_NO = b.CUST_AC_NO 
AND  a.RECORD_STAT = 'O'
AND a.AC_STAT_NO_DR = 'Y'
AND a.LCY_CURR_BALANCE >= '5000000'

UNION 

SELECT a.AC_DESC, a.CUST_AC_NO, a.CUST_NO, c.REASON_FOR_PND, a.AC_STAT_NO_DR AS PND_STAT, a.LCY_CURR_BALANCE
FROM STTM_CUST_ACCOUNT a, quality.ccpnd_detail c
WHERE a.CUST_AC_NO= c.ACCOUNT_NO
AND  a.RECORD_STAT = 'O'
AND a.AC_STAT_NO_DR = 'Y'
AND a.LCY_CURR_BALANCE >= '5000000';


SELECT * FROM PND_ENEH
