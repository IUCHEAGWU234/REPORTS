CREATE TABLE STAFF_LCY NOLOGGING NOCACHE PARALLEL(DEGREE 200) AS 
SELECT CUST_AC_NO AS AC_NO, AC_DESC AS AC_NAME, b.TRN_DT AS TRAN_DATE, 
b.TRN_CODE as TRAN_CODE, b.LCY_AMOUNT, CCY, DRCR_IND,
boengprod.BRG_ADDL_TEXT_tj(b.trn_ref_no,b.event_sr_no,b.module,b.ac_entry_sr_no)TRAN_DESCR
FROM STTM_CUST_aCCOUNT a, ACVW_ALL_AC_ENTRIES b
WHERE a.CUST_aC_NO = b.AC_NO
AND RECORD_STAT = 'O'
AND a.AUTH_STAT =  'A'
AND A.AUTH_ID <> 'NULL'
AND AC_CCY = 'NGN'
AND ACCOUNT_CLASS IN 
('GHSFFG','GHCATF','GHSTAF','GHSTDM','GHSGBS', 'GHSTDC','GHSFFC');
